65 def _parse_command(self, input_str: str) -> dict:
66 """
67 @brief Zerlegt Benutzereingaben in strukturierte Befehle
68 @param input_str Roh-Eingabe des Nutzers
69 @return Dictionary mit Schlüsseln: 'type', 'target', 'content'
70 @retval None bei ungültigen Befehlen
71
72 @details Unterstützte Befehle:
73 - join: Beitritt zum Netzwerk
74 - who: Zeigt bekannte Peers an
75 - msg <peer> <text>: Sendet Textnachricht
76 - img <peer> <pfad>: Sendet Bild
77 - leave: Verlässt das Netzwerk
78 - config: Zeigt Konfiguration
79 - exit: Beendet die Anwendung
80 """
81 parts = input_str.split(maxsplit=1)
82 if not parts:
83 return None
84
85 cmd_type = parts[0].lower()
86 command = {'type': cmd_type}
87
88 if cmd_type in ('msg', 'img'):
89 # Erwartet Format: <cmd> <target> <content>
90 subparts = parts[1].split(maxsplit=1) if len(parts) > 1 else []
91 if len(subparts) < 2:
92 print("Ungültige Eingabe: Ziel und Inhalt benötigt.")
93 return None
94 command['target'] = subparts[0]
95 command['content'] = subparts[1]
96 elif cmd_type in ('join', 'leave', 'who', 'exit', 'config'):
97 # Keine weiteren Argumente
98 pass
99 else:
100 print(f"Unbekannter Befehl: {cmd_type}")
101 return None
102
103 return command
