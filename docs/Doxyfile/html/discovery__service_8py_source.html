65 def _listen_udp(self):
66 """
67 @brief Lauscht kontinuierlich auf UDP-Broadcasts
68 @details Verarbeitet eingehende SLCP-Befehle
69 """
70 while self.running:

81 def _handle_slcp_command(self, command: str, source_ip: str):
82 """
83 @brief Verarbeitet SLCP-Discovery-Befehle
84 @param command SLCP-kodierter Befehl
85 @param source_ip Ursprungs-IP des Befehls
86
87 @details Unterstützte Befehle:
88 - JOIN:<username>:<tcp_port>
89 - LEAVE:<username>
90 - WHO
91 - KNOWUSERS:<json_peer_list>
92 """
93 if command.startswith("JOIN:"):
94 parts = command.split(':')
95 if len(parts) >= 3:
96 user = parts[1]
97 try:
98 tcp_port = int(parts[2])
99 # Aktualisiere Peer-Liste
100 self.peers[user] = {
101 'ip': source_ip,
102 'port': tcp_port,
103 'last_seen': time.time()
104 }
105 print(f"Neuer Peer: {user}@{source_ip}:{tcp_port}")
106 except ValueError:
107 pass
108
109 elif command == "WHO":
110 # Sende Liste der bekannten Peers
111 known_users = {user: data for user, data in self.peers.items()
112 if user != self.username}
113 response = f"KNOWUSERS:{json.dumps(known_users)}"
114 self.udp_socket.sendto(response.encode(), (source_ip, self.udp_port))
115
116 elif command.startswith("KNOWUSERS:"):
117 try:
118 json_str = command.split(':', 1)[1]
119 peer_data = json.loads(json_str)
120 current_time = time.time()
121
122 for user, data in peer_data.items():
123 # Aktualisiere nur, wenn der Peer nicht wir selbst sind
124 if user != self.username:
125 self.peers[user] = {
126 'ip': data['ip'],
127 'port': data['port'],
128 'last_seen': current_time
129 }
130 except (IndexError, json.JSONDecodeError):
131 pass
132
133 elif command.startswith("LEAVE:"):
134 user = command.split(':')[1]
135 if user in self.peers:
136 del self.peers[user]
137 print(f"Peer verlassen: {user}")
138
139 def _broadcast_presence(self):
140 """
141 @brief Sendet regelmäßig JOIN-Broadcasts

143 """
144 while self.running:

149 def _cleanup_peers(self):
150 """
151 @brief Entfernt inaktive Peers aus der Liste
152 @details Prüft regelmäßig die letzte Aktivität
153 """
154 while self.running:
155 current_time = time.time()
156 to_remove = []
157 for user, data in list(self.peers.items()):
158 if current_time - data['last_seen'] > self.PEER_TIMEOUT:
159 to_remove.append(user)
160 for user in to_remove:
161 del self.peers[user]
162 print(f"Peer timeout: {user}")
